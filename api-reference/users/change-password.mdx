---
title: "Change password"
api: "POST /api/v1/users/me/change-password"
description: "Change the authenticated user's password. Requires the current password and a matching confirmation of the new one."
---

<ParamField header="X-Tenant-ID" type="string" required>
  Tenant identifier
</ParamField>

<ParamField body="currentPassword" type="string" required>
  Current password
</ParamField>

<ParamField body="newPassword" type="string" required>
  New password (min 8 chars, must include uppercase, lowercase, number, special char)
</ParamField>
