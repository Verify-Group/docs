</ParamField>

</ParamField>

<ParamField query="isEnabled" type="boolean">
  Filter locally enabled or disabled sources
</ParamField>

<ParamField query="shortName" type="string">
  Case-insensitive exact match on sanctions.io short_name
</ParamField>
